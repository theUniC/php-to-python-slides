Exceptions
										
<?php

class DivisionByZeroException extends Exception
{
}

class NegativeDenominatorException extends Exception
{
}

function inverse(int $x): float {
	if (!$x) {
		throw new DivisionByZeroException('Division by zero.');
	}

	if ($x < 0) {
		throw new NegativeDenominatorException('Negative number provided');
	}

    return 1/$x;
}

try {
    echo inverse(5) . "\n";
    echo inverse(0) . "\n";
    echo inverse(-1) . "\n";
} catch (DivisionByZeroException | NegativeDenominatorException $e) {
    echo 'Caught exception: ',  $e->getMessage(), "\n";
} finally {
	echo "Program ended";
}

from typing import Union, final
from abc import ABC

class Game(ABC):
	def __init__(self):
		self.players_count = 0

	@abstractmethod
	def initialize_game(self) -> None:
		pass

	@abstractmethod
	def make_play(self, player: int) -> None:
		pass

	@abstractmethod
	def end_of_game(self) -> bool:
		pass

	@abstractmethod
	def print_winner(self) -> None:
		pass

	@final
	def play_one_game(self, players_count: int) -> None:
		self.players_count = players_count
		initialize_game()
		j = 0
		while not end_of_game():
			make_play(j)
			j = (j + 1) % players_count
		print_winner()

<?php

$list = [];

$chunks = array_chunk($list, 2); // [[1, 2], [3, 4]]

$intersection = array_intersect($list, [2, 3, 6]); // [2, 3]
$filtered = array_filter($list, static fn (int $item): bool => $item >= 2); // [2, 3, 4]
$times2 = array_map($list, static fn (int $item): int => $item * 2); // [2, 4, 6, 8]
$merge = array_merge($list, [5, 6, 7, 8]); // [1, 2, 3, 4, 5, 6, 7, 8]
$last = array_pop($list); // 4
$sum = array_reduce($list, static function (int $carry, int $item): int {
	$carry += $item;
	return $carry;
}); // 6

									
								
										
from functools import reduce

list = []
list.append(1)
list.append(2)
list.append(3)
list.append(4)

chunks = [list[i:i + 2] for i in range(0, len(list), 2)] # [[1, 2], [3, 4]]
diff = list(set(list) ^ set([1, 2, 3, 4, 5])) # [5]
intersection = list(set(list) & set([2, 3, 6])) # [2, 3]
filtered = filter(lambda item: item >= 2, list) # [2, 3, 4]
times2 = map(lambda item: item * 2, list) # [2, 4, 6, 8]
times2 = [item * 2 for item in list] # [2, 4, 6, 8]
merge = list + [5, 6, 7, 8] # [1, 2, 3, 4, 5, 6, 7, 8]
last = list.pop() # 4
sum = reduce(lambda carry, item: carry + item, list) # 6
